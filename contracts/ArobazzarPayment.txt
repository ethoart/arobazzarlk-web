pragma solidity ^0.8.0;

/**
 * @title Arobazzar Payment Forwarder V3
 * @dev Securely accepts payments and auto-forwards them to a specific recipient.
 *      DEPLOYMENT INSTRUCTION: Enter the destination wallet address in the constructor when deploying.
 */

interface IERC20 {
    function transferFrom(address sender, address recipient, uint256 amount) external returns (bool);
}

contract ArobazzarPaymentV3 {
    // The wallet that will actually receive the funds
    address public recipient;

    event PaymentReceived(address indexed payer, uint256 amount, string currency, string orderId, string details);

    // Run once on deployment. Input your main wallet address here.
    constructor(address _recipient) {
        require(_recipient != address(0), "Invalid recipient address");
        recipient = _recipient;
    }

    /**
     * @notice Pay with ETH (Native Coin)
     * Funds are immediately forwarded to the recipient wallet.
     */
    function payETH(string calldata orderId, string calldata details) external payable {
        require(msg.value > 0, "Amount must be > 0");
        
        // Auto-forward ETH to the recipient
        (bool success, ) = payable(recipient).call{value: msg.value}("");
        require(success, "ETH Forwarding failed");

        emit PaymentReceived(msg.sender, msg.value, "ETH", orderId, details);
    }

    /**
     * @notice Pay with ERC20 Tokens (USDC, USDT)
     * Tokens are transferred directly from the customer to the recipient wallet.
     * Note: Customer must Approve this contract address first.
     */
    function payToken(address token, uint256 amount, string calldata orderId, string calldata details) external {
        require(amount > 0, "Amount must be > 0");
        
        // Transfer tokens directly from Customer -> Recipient
        // The contract does not hold the tokens.
        bool success = IERC20(token).transferFrom(msg.sender, recipient, amount);
        require(success, "Token Transfer failed. Check allowance.");

        emit PaymentReceived(msg.sender, amount, "ERC20", orderId, details);
    }

    /**
     * @notice Update the recipient address if you change your wallet.
     * Only the current recipient can call this.
     */
    function updateRecipient(address _newRecipient) external {
        require(msg.sender == recipient, "Unauthorized: Only current recipient can update");
        require(_newRecipient != address(0), "Invalid address");
        recipient = _newRecipient;
    }
}
